    Create React hook for contract interactions:

    ```typescript
    export function useContract() {
      const [isPending, setIsPending] = useState(false);
      const [error, setError] = useState<string | null>(null);

      const createGame = useCallback(async (roomId: string, betAmount: number, maxPlayers: number) => {
        setIsPending(true);
        setError(null);
        try {
          const result = await contractService.createGame(roomId, betAmount, maxPlayers);
          return result;
        } catch (err) {
          setError(err.message);
          throw err;
        } finally {
          setIsPending(false);
        }
      }, []);

      const joinGame = useCallback(async (roomId: string, betAmount: number) => {
        setIsPending(true);
        setError(null);
        try {
          const result = await contractService.joinGame(roomId, betAmount);
          return result;
        } catch (err) {
          setError(err.message);
          throw err;
        } finally {
          setIsPending(false);
        }
      }, []);

      return { createGame, joinGame, isPending, error };
    }
    ```

    The hook wraps the contract service with React state for loading and error handling.
    Note: distributePrizes is NOT included here - it's in Plan 06 (GameResults wiring).
